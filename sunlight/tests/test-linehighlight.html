SELECT
    f.foo `foo`,
    f.bar,
    f.name AS `foo.name`,
    b.baz
FROM foo f
INNER JOIN bar b
    ON b.id = f.bar_id
LEFT OUTER JOIN baz
    ON baz.id = f.foo
    AND baz.id IS NOT NULL
GROUP BY f.name
HAVING COUNT(f.name) < 10
WHERE LENGTH(f.foo) < 5
AND LEFT(f.foo) = 'f'
